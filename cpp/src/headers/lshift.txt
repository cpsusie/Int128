ssi::char8_tests::uint128 ssi::char8_tests::lshift_u128(uint128 shift_me, int shift_amount)
{
	assert(shift_amount > -1 && shift_amount < 128);
	auto ret = uint128{};
	if (shift_amount >= 64)
	{
		ret.high = shift_me.low;
		ret.low = 0;
		ret.high <<= (shift_amount - 64);
	}
	else
	{
		ret.high = __shiftleft128(shift_me.low, shift_me.high, static_cast<unsigned char>(shift_amount));
		ret.low = shift_me.low << shift_amount;
	}
	return ret;
}
